      } else if (Syntax[type][constructor].length === 1) {
        exports[constructor] = (field1) => [constructor, field1];
      } else if (Syntax[type][constructor].length === 2) {

      } else if (Syntax[type][constructor].length === 3) {

      } else {

      }
    });
  });
};
